def grade_one(client, conn: sqlite3.Connection, expression: str, run_id: str) -> dict:
    """
    Grade a single expression: validate → simulate → IS checks → (if survivor) POST /check → persist.
    Returns a result dict with keys: expression, status, is_survivor, checks, self_corr, prod_corr.
    """

def grade_many(client, conn: sqlite3.Connection, expressions: list[str],
               run_id: str, max_workers: int = 1) -> list[dict]:
    """
    Grade a list of expressions. max_workers ≤ 3.
    Sequential (max_workers=1) is the Phase 1 default.
    """

def trigger_correlation_check(client, alpha_id: str) -> None:
    """POST /alphas/{alpha_id}/check to trigger server-side correlation computation."""

def poll_correlation(client, alpha_id: str, timeout: int = 300, interval: int = 15) -> dict:
    """
    Poll GET /alphas/{alpha_id} until SELF_CORRELATION + PROD_CORRELATION leave PENDING.
    Returns {SELF_CORRELATION: check_dict, PROD_CORRELATION: check_dict}.
    Raises TimeoutError if not resolved within timeout seconds.
    """
