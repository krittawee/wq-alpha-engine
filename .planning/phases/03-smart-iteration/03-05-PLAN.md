    Private _rank_best(alpha_ids: list[str], conn: sqlite3.Connection) -> Optional[str]:
    - For each alpha_id, read sharpe + fitness from alphas
    - Rank by Sharpe descending (Claude's discretion per D-20 — Sharpe is primary ranking metric)
    - Return best alpha_id or None if list is empty

    Implement hunt(client, db_path: str = "alpha_kb.db", max_depth: int = 2, max_sims: int = 30) -> dict:
    - conn = db.init_db(db_path); try/finally: conn.close()
    - run_id = str(uuid.uuid4())[:8]
    - One-time backfill before loop: selfcorr.backfill_active_pnl(client, conn, db_path)
      (fetches PnL for ACTIVE alphas with pnl_path IS NULL — builds reference set for selfcorr)
    - Capture diversity BEFORE: diversity_before = fsa.diversity_metric(conn)
      (snapshot of structural concentration in current PASS alpha pool — criterion 4 baseline)
    - avoid_motifs = fsa.mine_frequent_motifs(conn)
    - thesis = researcher.build_thesis(conn, avoid_motifs=avoid_motifs)
    - candidates = ideator.generate_candidates(conn, thesis)
    - cand_dicts = [{"expression": c.get("expression",""), **c} for c in candidates]
    - filtered = fsa.filter_candidates(cand_dicts, avoid_motifs)
    - queue = [c["expression"] for c in ideator.queueable(filtered)]
    - Gen 0: results = grade.grade_many(client, conn, queue, run_id, max_workers=3, db_path=db_path)
    - sims_used = len(queue)
    - best_submittable = None; best_near = []

    Loop for gen in range(max_depth):
      a) Reclassify all results with alpha_id:
         for r in results where r.get("alpha_id"):
           status, _ = editor.classify_from_checks(r["alpha_id"], conn)
           if status == "near": near_ids.append(r["alpha_id"])
           if status == "pass": pass_ids.append(r["alpha_id"])
      b) Update best_submittable = _rank_best(pass_ids + previous_pass_ids, conn) (accumulate across gens)
      c) best_near.extend(near_ids)
      d) Stop conditions (D-16): if sims_used >= max_sims: break; if not near_ids: break
      e) Collect mutations from all near + fail alphas:
         all_mutations = []
         for each r with alpha_id, if status in ("near","fail"):
             ed = editor.diagnose_and_mutate(r["alpha_id"], conn, avoid_motifs=avoid_motifs)
             all_mutations.extend(ed["mutations"])
      f) Filter + FSA-steer mutations:
         Mutations returned by editor.diagnose_and_mutate are ALREADY validated, deduped,
         and PRE-INSERTED as status='queued' stubs. Do NOT re-apply a blanket db.expr_exists
         filter to them — that would drop every editor-returned mutation (all are now in DB).
         Instead, keep a mutation if it is not yet in a terminal/graded state:
           def _is_passable(conn, expr):
               existing_id = db.expr_exists(conn, expr)
               if existing_id is None:
                   return True  # brand-new expression (not from editor path)
               row = conn.execute(
                   "SELECT status FROM alphas WHERE alpha_id=? LIMIT 1",
                   (existing_id,)
               ).fetchone()
               return row is not None and row[0] == "queued"

           all_mutations = [m for m in all_mutations if _is_passable(conn, m)]
         Then apply FSA structural filter on top:
           mut_dicts = [{"expression": m} for m in all_mutations]
           mut_dicts = fsa.filter_candidates(mut_dicts, avoid_motifs)
           queue_next = [d["expression"] for d in mut_dicts][: max_sims - sims_used]
           if not queue_next: break
      g) grade_many for next generation:
         The mutations in queue_next are already in the DB as status='queued' stubs with
         parent_alpha_id set — this was done by editor.diagnose_and_mutate before returning.
         grade_many receives these expressions and grades them in place; each worker finds
         the existing row via db.expr_exists and updates it (status, sharpe, fitness, etc.).
         hunt() does NOT patch parent_alpha_id post-grading — lineage is guaranteed at
         pre-insert time by editor.diagnose_and_mutate per plan 03-01.
      h) sims_used += len(queue_next)
      i) results = grade_many results for next gen

    - Capture diversity AFTER final generation: diversity_after = fsa.diversity_metric(conn)
    - Log: print(f"[hunt] diversity before: {diversity_before['top_motif_share']:.2%} | after: {diversity_after['top_motif_share']:.2%}")

    Return: {"best_submittable": best_submittable, "best_near": best_near, "sims_used": sims_used,
             "run_id": run_id, "generations": gen+1,
             "diversity_before": diversity_before, "diversity_after": diversity_after}
